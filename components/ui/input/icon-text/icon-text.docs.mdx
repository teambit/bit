import { useState } from 'react';
import { Icon } from '@teambit/design.elements.icon';
import { IconTextInput } from './icon-text';

### Usages

IconTextInput component based on Text component.  
Supports all props from native html input.  
Size support with em, the container of the component can use `font-size` to change the size of the component.

### Icon and Text

```js live
() => {
  const [value, setValue] = useState('');
  return (
    <IconTextInput placeholder="search..." icon={<Icon of="discovery" />} onChange={(e) => setValue(e.target.value)} />
  );
};
```
