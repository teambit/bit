import { MemoryRouter, Route, useLocation } from 'react-router-dom';
import { Link } from './link';
import { NavLink } from './nav-link';

React Router 5 implementation of our isomorphic link. Read more [here](https://bit.dev/teambit/base-react/navigation/link).  
Use

> ⚠️ Do not use this link directly! Use [base link](https://bit.dev/teambit/base-react/navigation/link) or [designed link](https://bit.dev/teambit/design/ui/navigation/link).  
> Use [routing-adapter](https://bit.dev/teambit/ui-foundation/ui/navigation/react-router/routing-adapter) to inject this component as the concrete implementation for Link.

### Usage

The link component behaves like a native `<a>` tag. It does assume a React Router `<Router>` context is available.

```tsx
<Link href="/pricing">pricing page</Link>
```

Here is an example of this Link component in action, used directly with React Router. (in a real world situation, we would use it through the Routing Adapter)  
Try it out!

```tsx live
() => {
  const Component = () => {
    const location = useLocation();

    return (
      <div style={{ maxWidth: 500 }}>
        {/* current location: */}
        <div>
          location: <br />
          {JSON.stringify(location, undefined, 1)}
        </div>

        {/* links: */}
        <div>
          <NavLink href="/path/one?what=is" state={{ some: 'state' }} activeStyle={{ fontWeight: 'bold' }}>
            /path/one
          </NavLink>
          <br />
          <NavLink href="/path/two" activeStyle={{ fontWeight: 'bold' }}>
            /path/two
          </NavLink>
        </div>

        <br />
        {/* react router matches the current location */}
        <Route path="/path/one">First path</Route>
        <Route path="/path/two">Second path</Route>
      </div>
    );
  };

  return (
    // a react router wrapper is required
    <MemoryRouter>
      <Component />
    </MemoryRouter>
  );
};
```
