### Component usage

Use your component in any mdx file:

```js

```

```js

import { Generator } from '@teambit/aspect-docs.generator';

<ThemeCompositions>
  <MDXLayout>
    <Generator />
  </MDXLayout>
</ThemeCompositions>;
```
